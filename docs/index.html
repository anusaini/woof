woof

🐶 command line applications made as easy as fetch

             npm install woof 
          
const cli = woof(`
    Usage
      $ foo [input]

    Options
      --unicorn, -u [type]    Include a unicorn [rainbow|sea]

    Examples

      $ foo --unicorn rainbow
      🌈 🦄 🌈
`, {
  flags: {
    unicorn: {
      type: 'string',
      alias: 'u',
      default: 'rainbow'
    }
  }
});
